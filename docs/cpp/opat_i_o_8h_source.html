719template <typename T>
720T swap_bytes(T value) {
721 static_assert(std::is_trivially_copyable_v<T>, "swap_bytes only supports trivial types.");
722 T result;
723 uint8_t* src = reinterpret_cast<uint8_t*>(&value);
724 uint8_t* dest = reinterpret_cast<uint8_t*>(&result);
725 for (size_t i = 0; i < sizeof(T); i++) {
726 dest[i] = src[sizeof(T) - 1 - i];
727 }
728 return result;
729}
