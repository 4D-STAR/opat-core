518template <typename T>
519T swap_bytes(T value) {
520 static_assert(std::is_trivially_copyable<T>::value, "swap_bytes only supports trivial types.");
521 T result;
522 uint8_t* src = reinterpret_cast<uint8_t*>(&value);
523 uint8_t* dest = reinterpret_cast<uint8_t*>(&result);
524 for (size_t i = 0; i < sizeof(T); i++) {
525 dest[i] = src[sizeof(T) - 1 - i];
526 }
527 return result;
528}
